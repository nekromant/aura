91  int flags, off_t offset, unsigned char **ptr, int *map_fd)
92 {
93  int ret;
94  struct ion_fd_data data = {
95  .handle = handle,
96  };
97 
98  if (map_fd == NULL)
99  return -EINVAL;
100  if (ptr == NULL)
101  return -EINVAL;
102 
103  ret = ion_ioctl(fd, ION_IOC_MAP, &data);
104  if (ret < 0)
105  return ret;
106  *map_fd = data.fd;
107  if (*map_fd < 0) {
108  ALOGE("map ioctl returned negative fd\n");
109  return -EINVAL;
110  }
111  *ptr = mmap(NULL, length, prot, flags, *map_fd, offset);
112  if (*ptr == MAP_FAILED) {
113  ALOGE("mmap failed: %s\n", strerror(errno));
114  return -errno;
115  }
116  return ret;
